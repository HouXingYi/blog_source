# 第一周 基本概念

## 总结

数据结构的定义与内容，算法的定义与内容，一个实例：最大子列和问题。

## 什么是数据结构

解决问题方法的效率和数据的组织方式有关。

组织方式：

1. 数据对象集的逻辑结构。线性，树状，图状。

2. 数据对象集在计算机中的物理存储结构。数组存储，链表存储。

抽象数据类型（数据对象集，操作集）

## 什么是算法

算法空间复杂度S(n)，时间复杂度T(n)。

时间复杂度计算方法：寻找执行次数t与问题的规模n的表达式，得出t>f(n)时算法退出，f(n)即为时间复杂度。意义为最深的执行次数（频度）与问题的规模n的关系。

## 应用实例：最大子列和问题

1. 直接穷举O(n^3)

2. 部分存储中间值的穷举O(n^2)

3. 分而治之O(NlogN)

4. 在线处理O(N)

#include<stdio.h> 

#define MaxTree 11
#define ElementType char
#define Tree int
#define Null -1

struct TreeNode {
    ElementType Element;//树结点的值，字符 
    Tree Left; 
    Tree Right;
} T1[MaxTree], T2[MaxTree];

// 创建树，树的结构为结构数组 
Tree BuildTree (struct TreeNode T[]) {
    int N, i = -1;
    int check[11];
    char cl, cr;
    scanf("%d", &N);
    getchar();
     
    for(i = 0; i < N; i++) {
		check[i] = 0;
	}
	
    if(!N) {
		return Null;
	}
	
    if(N) {
        for(i = 0; i < N; i++) {
            scanf("%c %c %c", &T[i].Element, &cl,&cr);
            getchar();
            
            // 左子树 
            if(cl != '-') {
                T[i].Left = cl - '0';
                check[T[i].Left] = 1;
            } else {
				T[i].Left = Null;
			} 
            
            // 右子树
            if(cr != '-') {
                T[i].Right = cr - '0';
                check[T[i].Right] = 1;
            } else {
				T[i].Right = Null;
			}
        }    
    }
    
    for (i = 0; i < N; i++) {
		if(!check[i]) {
			return i; // 没被check的即为root 
		} 
	} 
    
}

// 是否是同构对比 
int Isomorphic(Tree R1, Tree R2) {
	
	// 是同构 
    if(R1 == Null && R2 == Null) {
		return 1;
	} 
    
    // 不是同构
    if(R1 == Null && R2 != Null) {
		return 0;
	} 
    
    // 不是同构
    if(R1 != Null && R2 == Null) {
		return 0;
	} 
    
    // 不是同构
    if(T1[R1].Element != T2[R2].Element) {
    	return 0;	
	} 
    
    // 找遍每一种可能性 
    
    if(T1[R1].Left == Null && T2[R2].Left == Null) {
		return Isomorphic(T1[R1].Right, T2[R2].Right);
	}
    
    if(T1[R1].Left != Null && T2[R2].Left != Null && T1[T1[R1].Left].Element == T2[T2[R2].Left].Element) {
		return Isomorphic(T1[R1].Left, T2[R2].Left) && Isomorphic(T1[R1].Right, T2[R2].Right);
	} else {
		return Isomorphic(T1[R1].Right, T2[R2].Left) && Isomorphic(T1[R1].Left, T2[R2].Right);
	} 
    
}

int main() {
	
    Tree R1, R2;
    
    R1 = BuildTree(T1);
    
    R2 = BuildTree(T2);
    
    if (Isomorphic(R1,R2) ) {
		printf("Yes\n");
	} else {
		printf("No\n");
	}
	
    return 0;
}

#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<queue>

using namespace std;

#define MaxTree 10
#define Null -1

typedef int Tree;

// 树结构 
struct TreeNode {
    Tree Left;
    Tree Right;
} T [MaxTree]; // 结构数组
 
int N, check[MaxTree] = { 0 }, count = 0;

queue<int> q;

// 建树 
Tree BuildTree(struct TreeNode T[]){
    Tree Root=Null;
    int i;
    char cl,cr;
    scanf("%d",&N);
    if(N){
        for(i = 0; i < N; i++){
            scanf("\n%c %c", &cl, &cr);
            
            // 左子树 
            if(cl != '-'){
                T[i].Left = cl - '0';
                check[T[i].Left] = 1;
            } else {
				T[i].Left=Null;
			}
			
			// 右子树 
            if (cr != '-') {
                T[i].Right = cr - '0';
                check[T[i].Right] = 1;
            } else {
				T[i].Right=Null;
			}
        }   
        
        for(i=0;i<N;i++) {
			if(check[i]==0) break;
		}
		
        Root = i;
    }
    return Root;
}

// 层序遍历（利用队列） 
void countleaves(Tree Root){        //每碰到一个节点将它的左右孩子入队（若有的话），然后依次从队头取出判断是否为叶子节点
    Tree temp;
    if(Root==Null)  return;
    q.push(Root);          
    while(!q.empty()) {
        temp = q.front();
        q.pop();
        if(T[temp].Left == Null && T[temp].Right == Null) {    //如果没有左右孩子即为叶子节点，则输出
            if(count++ != 0){     //不是第一个节点的话前面输出空格
                putchar(' ');
            }
            printf("%d",temp);          
        }
        if(T[temp].Left != Null)      q.push(T[temp].Left);
        if(T[temp].Right != Null)     q.push(T[temp].Right);
    }
}
int main(){
    Tree R;
    R = BuildTree(T);
    countleaves(R);
    return 0;
}

3. Tree Traversals Again 是2014年秋季PAT甲级考试真题，稍微要动下脑筋，想通了其实程序很基础，建议尝试。

注意点：

考点：

书本109页，由两种遍历序列确定二叉树。先序序列可确定根节点（确定后到中序中找根），中序序列可确定左右子树的范围（再到先序中找）。

另有考点，先中后遍历的递归与非递归形式。

stringstream用于格式的转化（类似于sscanf）。

```
#include <iostream>
#include <stdio.h>
#include <vector>
#include <string>
#include <sstream>
#include <stack>
 
using namespace std;
 
int N, cur = 0;
vector<int> preOrder;
vector<int> inOrder;
 
// 树结构 
typedef struct TreeNode *Node;
struct TreeNode{
    int num;
    Node left, right;
    TreeNode(){
        left = NULL;
        right = NULL;
    }
};
 
// 根据值找index（在中序中寻找左右结点范围） 
int findRootIndex(int rootNum){ 
    for(int i = 0; i < N; i++){
        if(inOrder[i] == rootNum){
            return i;
        }
    }
    return -1;
}

// 树的创建 (通过先序和中序确定一棵树)
Node CreateTree(int left, int right){
    if(left > right) return NULL;
    int root = preOrder[cur]; // 先序确定根结点 
    cur++;
    int rootIndex = findRootIndex(root); // 中序确定左右结点的范围 
    Node T = new TreeNode();
    T->num = root;
    if(left != right) {
        T->left = CreateTree(left,rootIndex-1);
        T->right = CreateTree(rootIndex+1,right);
    }
    return T; 
}
 
bool firstOutPut = true;

void PostOrder(Node T){
    if(!T) return;
    PostOrder(T->left);
    PostOrder(T->right);
    if(firstOutPut){
        printf("%d", T -> num);
        firstOutPut = false;
    }else{
        printf(" %d", T -> num);
    }
}
 
 
int main()
{
    stringstream ss; // 用于将string转化为int 
    string Nstr;
    getline(cin, Nstr);
    ss << Nstr;
    ss >> N;
    ss.clear();
    string input;
    stack<int> stk;
    int value;
    
    for(int i = 0; i < N * 2; i++){
        getline(cin,input);
        if(input[1] == 'u'){ // push 
            string num = input.substr(5);
            ss << num;
            ss >> value;
            ss.clear();
            stk.push(value);
            preOrder.push_back(value); // 构造先序遍历序列
        }else{ // pop 
            value = stk.top();
            stk.pop();
            inOrder.push_back(value); // 构造中序遍历序列 
        }
    }
    Node T = CreateTree(0, N-1); // 通过先序序列与中序序列确定一棵树 
    PostOrder(T); // 后序遍历并打印结点 
    return 0;
}
```
